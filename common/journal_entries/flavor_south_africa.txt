je_great_trek = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"



    	complete = {	
  		custom_tooltip = {
			text = boer_republics_formed
			hidden_trigger = {
					        AND = {
					exists = c:TRN	
                    exists = c:ORA
                    exists = c:NAL
					exists = c:KLE
					exists = c:STE
                    }
			}
		}	
	}

	fail = {
		OR = {
			has_law = law_type:law_slave_trade
			has_law = law_type:law_legacy_slavery
		}
	}

    on_complete = {
		trigger_event = { id = journal_flavor_russia.1 } # Towards India
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_great_trek_status
			}
		}
	}
    should_be_pinned_by_default = yes
	weight = 100
}

je_autonomy = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"



    	complete = {	
GDP >= 1000000
	}

	fail = {
		c:SAF = {
			NOT = { is_subject_of = c:GBR }
		}
	}

    on_complete = {
		c:GBR = {
			create_diplomatic_pact = {
				country = c:SAF
				type = dominion
			}
		}
		if = {
			limit = { 
				exists = c:GBR
				exists = c:XHO
				c:GBR = {
					has_diplomatic_pact = {
						who = c:XHO
						type = puppet
						is_initiator = yes
					}
				} 
			}
			c:SAF = {
            annex = c:XHO
        	}
		}
			
		
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_great_trek_status
			}
		}
	}
    should_be_pinned_by_default = yes
	weight = 1000
}

je_natalia_republic = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

on_monthly_pulse = {
	random_events = {
				4 = journal_flavor_south_africa.1
				1 = journal_flavor_south_africa.1
			}	
}
				
	
	

	complete = {	
  		custom_tooltip = {
			text = nal_exists
			hidden_trigger = {
					exists = c:NAL
			}
		}	
	}

	fail = {
		OR = {
			has_law = law_type:law_slave_trade
			has_law = law_type:law_legacy_slavery
	
			any_state = {
				state_region = s:STATE_NATALIA
				owner = { 
					NOT = { 
						OR = { 
							is_country_type = decentralized
							is_country_type = unrecognized
						}
					} 
					owns_entire_state_region = STATE_NATALIA
				}
			}
		}
	}

	weight = 100
}
