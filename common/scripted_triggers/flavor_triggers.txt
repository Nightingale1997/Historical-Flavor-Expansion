coa_def_greater_subject_trigger = {
	OR = {
		NOT = { exists = scope:target }
		AND = {
			exists = scope:target
			scope:target = { 
				OR = {
					is_subject_type = subject_type_puppet
					is_subject_type = subject_type_vassal
                    is_subject_type = subject_type_dominion
				}
			}
		}
	}
}

coa_def_russian_subject_trigger = {
	OR = {
		NOT = { exists = scope:target }
		AND = {
			exists = scope:target
			scope:target = { 
			exists = c:RUS
			c:RUS = {
				has_diplomatic_pact = {
					who = scope:target 
					type = puppet
					is_initiator = yes
				}
			}
			}
		}
	}
}
